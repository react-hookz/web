# useNetworkState

Tracks the state of browser's network connection.

> As of the standard, it is not guaranteed that browser connected to the _Internet_, it only
> guarantees the network connection.
> [[MDN Docs]](https://developer.mozilla.org/en-US/docs/Web/API/NavigatorOnLine/onLine)

## Reference

```ts
export interface IUseNetworkState {
	online: boolean | undefined;
	previous: boolean | undefined;
	since: Date | undefined;
	downlink: number | undefined;
	downlinkMax: number | undefined;
	effectiveType: 'slow-2g' | '2g' | '3g' | '4g' | undefined;
	rtt: number | undefined;
	saveData: boolean | undefined;
	type:
		| 'bluetooth'
		| 'cellular'
		| 'ethernet'
		| 'none'
		| 'wifi'
		| 'wimax'
		| 'other'
		| 'unknown'
		| undefined;
}

export function useNetworkState(initialState?: InitialState<IUseNetworkState>): IUseNetworkState;
```
